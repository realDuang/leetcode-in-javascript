var reconstructQueue = function(people) {
  people.sort((a, b) => (a[0] === b[0] ? a[1] - b[1] : b[0] - a[0]));
  const res = [];
  people.forEach(ele => {
    res.splice(ele[1], 0, ele);
  });
  return res;
};
