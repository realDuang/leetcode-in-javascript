var generateParenthesis = function(n) {
  const res = [];
  helper('', n, n, res);
  return res;
};

function helper(str, left, right, res) {
  if (left < 0 || right < 0 || left > right) return;
  if (left === 0) {
    str += ')'.repeat(right);
    res.push(str);
    return;
  }

  helper(str + '(', left - 1, right, res);
  helper(str + ')', left, right - 1, res);
}
