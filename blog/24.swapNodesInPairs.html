var swapPairs = function(head) {
  const dummy = new ListNode(null);
  dummy.next = head;
  let pre = dummy;
  while (pre.next && pre.next.next) {
    const cur = pre.next;
    const next = cur.next;

    pre.next = next;
    cur.next = next.next;
    next.next = cur;

    pre = cur;
  }
  return dummy.next;
};

function ListNode(val) {
  this.val = val;
  this.next = null;
}
