var isValid = function(s) {
  if (s.length % 2 !== 0) return false;
  const stack = [];
  const left = new Set(['{', '[', '(']);
  const right = new Set(['}', ']', ')']);
  const map = {
    '}': '{',
    ']': '[',
    ')': '('
  };
  for (let i = 0; i < s.length; i++) {
    if (left.has(s[i])) {
      stack.push(s[i]);
    } else if (right.has(s[i])) {
      if (stack.length === 0) {
        return false;
      }
      if (map[s[i]] !== stack[stack.length - 1]) {
        return false;
      } else {
        stack.pop();
      }
    } else {
      return false;
    }
  }
  if (stack.length === 0) {
    return true;
  }
  return false;
};
