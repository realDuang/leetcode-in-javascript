function threeSum(nums: number[]): number[][] {
  nums = nums.sort((a, b) => a - b);

  const res: number[][] = [];
  for (let i = 0; i < nums.length - 2; i++) {

    if (min > 0) break;
    // å»æ‰é‡å¤æƒ…å†µ
    if (i > 0 && min === nums[i - 1]) continue;

    const target = 0 - min;
    // æ¥ä¸‹æ¥å°±æ˜¯ 2Sum é—®é¢˜äº†
    let left = i + 1;
    let right = nums.length - 1;
    while (left < right) {
      if (nums[left] + nums[right] === target) {
        res.push([min, nums[left], nums[right]]);
        // å»é™¤é‡å¤æƒ…å†µ
        while (left < right && nums[left + 1] === nums[left]) left += 1;
        while (left < right && nums[right - 1] === nums[right]) right -= 1;

        // æŒ‡é’ˆç§»åŠ¨åˆ°ä¸‹ä¸€ç»„æƒ…å†µ
        left += 1;

      } else if (nums[left] + nums[right] > target) {

      } else {
        left += 1;
      }
    }
  }
  return res;
}
