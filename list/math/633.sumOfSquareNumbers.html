var judgeSquareSum = function(c) {
  let lp = 0;
  let rp = Math.floor(Math.sqrt(c));
  while (lp <= rp) {
    const res = lp * lp + rp * rp;
    if (res === c) {
      return true;
    } else if (res > c) {
      rp--;
    } else {
      lp++;
    }
  }
  return false;
};
