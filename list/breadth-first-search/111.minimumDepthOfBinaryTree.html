var minDepth = function(root) {
  if (!root) return 0;

  let res = 1;
  const queue = [root];

  while (queue.length !== 0) {
    // è¿™é‡Œæå‰ç®—å‡ºqueueçš„é•¿åº¦è¿›è¡Œå¾ªç¯ï¼Œæ³¨æ„ä¸€å®šä¸è¦å°†lengthå†™åœ¨å¾ªç¯é‡ŒåŠ¨æ€åˆ¤æ–­ï¼Œå› ä¸ºqueueçš„é•¿åº¦æ˜¯ä¼šå˜çš„ï¼
    const layerNum = queue.length;
    for (let i = 0; i < layerNum; i++) {
      const cur = queue.shift();
      if (cur.left === null && cur.right === null) {
        return res;
      }

      if (cur.left !== null) {
        queue.push(cur.left);
      }
      if (cur.right !== null) {
        queue.push(cur.right);
      }
    }
    res++;
  }
  return res;
};
