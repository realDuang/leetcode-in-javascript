var solveNQueens = function(n) {
  const res = [];
  const map = Array(n)

  backtrack(map, 0);
  return res;

  function backtrack(map, row) {
    if (row >= n) {
      res.push(map.map(arr => arr.join('')));

function isValid(map, row, col) {
