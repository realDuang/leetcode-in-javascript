function permute(nums: number[]): number[][] {
  const res: number[][] = [];
  const path: number[] = [];
  backtrack(nums);
  return res;

  function backtrack(rest: number[]) {
    if (rest.length === 0) {
      res.push([...path]);
      return;
    }
    rest.forEach((num, index) => {
      // åšé€‰æ‹©

  }
}
