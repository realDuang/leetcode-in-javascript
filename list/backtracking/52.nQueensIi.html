var totalNQueens = function(n) {
  let res = 0;
  const map = Array(n)
    .fill('')
    .map(x => Array(n).fill('.'));
  backtrack(map, 0);
  return res;

  function backtrack(map, row) {
    if (row >= n) {
      res += 1;
      return;
    }

    for (let col = 0; col < n; col++) {
      if (isValid(map, row, col)) {
        map[row][col] = 'Q';
        backtrack(map, row + 1);
        map[row][col] = '.';
      }
    }
  }

  function isValid(map, row, col) {
    let i = row - 1;
    let topLeft = col - 1;
    let topRight = col + 1;
    while (i >= 0) {
      if (map[i][topLeft] === 'Q' || map[i][col] === 'Q' || map[i][topRight] === 'Q') return false;
      i--;
      topLeft--;
      topRight++;
    }

    return true;
  }
};
