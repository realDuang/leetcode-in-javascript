function twoSum(nums: number[], target: number): number[] {
  const hashMap: Record<number, number> = {};

  for (let i = 0; i < nums.length; i++) {
    const rest = target - nums[i];
    if (hashMap[rest] !== undefined) {
      return [i, hashMap[rest]];
    }
    hashMap[nums[i]] = i;
  }
  return [];
}
