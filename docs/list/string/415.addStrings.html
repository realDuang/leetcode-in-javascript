var addStrings = function(num1, num2) {
  let i = num1.length - 1;
  let j = num2.length - 1;
  if (i < j) num1 = '0'.repeat(j - i) + num1;
  if (i > j) num2 = '0'.repeat(i - j) + num2;

  let res = '';
  let cnt = 0;

  for (let k = Math.max(i, j); k >= 0; k--) {
    const temp = Number(num1[k]) + Number(num2[k]) + cnt;
    cnt = temp > 9 ? 1 : 0;
    res = (temp % 10) + res;
  }
  if (cnt === 1) res = '1' + res;
  return res;
};
