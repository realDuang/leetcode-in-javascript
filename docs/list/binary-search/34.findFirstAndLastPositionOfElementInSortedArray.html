function searchRange(nums: number[], target: number): number[] {
  const res: number[] = [-1, -1];
  if (nums.length === 0) return res;

  // æ±‚å·¦è¾¹ç•Œ
  let left = 0;
  let right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] < target) {
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  if (left < nums.length && nums[left] === target) {
    res[0] = left;
  } else {
    res[0] = -1;
  }

  // æ±‚å³è¾¹ç•Œ
  left = 0;
  right = nums.length - 1;
  while (left <= right) {
    const mid = Math.floor((left + right) / 2);
    if (nums[mid] > target) {
      right = mid - 1;
    } else {
      left = mid + 1;
    }
  }
  if (right >= 0 && nums[right] === target) {
    res[1] = right;
  } else {
    res[1] = -1;
  }

  return res;
}
